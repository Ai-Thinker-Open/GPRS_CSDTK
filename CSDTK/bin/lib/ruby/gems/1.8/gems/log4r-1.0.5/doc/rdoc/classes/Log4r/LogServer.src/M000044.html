    def initialize(_fullname, _uri, _level=nil, 
                   _additive=true, _trace=false, &accept)
      super(_fullname, _level, _additive, _trace)
      @uri = _uri
      start_server(_uri, accept)
      Logger.log_internal {"LogServer started at #{@uri}"}
    end
