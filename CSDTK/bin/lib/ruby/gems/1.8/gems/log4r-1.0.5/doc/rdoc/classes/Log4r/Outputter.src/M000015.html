    def self.[](name)
    out = @@outputters[name]
      if out.nil?
        return case name
        when 'stdout' then StdoutOutputter.new 'stdout'
        when 'stderr' then StderrOutputter.new 'stderr'
        else nil end
      end          
      out
    end
